  const [copied, setCopied] = useState(false);

export const Poll = () => {
  const [votes, setVotes] = useState({ react: 0, vue: 0, svelte: 0 });
  const [userVoted, setUserVoted] = useState(null);

  const vote = (framework) => {
    if (userVoted) return;
    setVotes((prev) => ({ ...prev, [framework]: prev[framework] + 1 }));
    setUserVoted(framework);
  };

  const total = votes.react + votes.vue + votes.svelte;

  return (
    <div className='rounded-lg border border-gray-200 bg-white p-6 shadow-lg dark:border-gray-700 dark:bg-gray-800'>
      <h4 className='mb-4 text-lg font-semibold text-gray-800 dark:text-white'>
        ðŸ“Š Which framework do you prefer?
      </h4>

      <div className='space-y-3'>
        {Object.entries(votes).map(([framework, count]) => {
          const percentage = total > 0 ? (count / total) * 100 : 0;
          return (
            <div key={framework} className='space-y-2'>
              <div className='flex items-center justify-between'>
                <button
                  onClick={() => vote(framework)}
                  disabled={userVoted}
                  className={`rounded px-4 py-2 text-sm font-medium capitalize transition-colors ${
                    userVoted === framework
                      ? 'bg-green-500 text-white'
                      : userVoted
                        ? 'cursor-not-allowed bg-gray-300 text-gray-500'
                        : 'bg-blue-500 text-white hover:bg-blue-600'
                  }`}
                >
                  {framework}
                </button>
                <span className='text-sm text-gray-600 dark:text-gray-400'>
                  {count} votes ({percentage.toFixed(1)}%)
                </span>
              </div>
              <div className='h-2 w-full rounded-full bg-gray-200 dark:bg-gray-700'>
                <div
                  className='h-2 rounded-full bg-blue-500 transition-all duration-500'
                  style={{ width: `${percentage}%` }}
                ></div>
              </div>
            </div>
          );
        })}
      </div>

      {userVoted && (
        <div className='mt-4 rounded border border-green-200 bg-green-50 p-3 dark:border-green-800 dark:bg-green-900/20'>
          <p className='text-sm text-green-800 dark:text-green-300'>
            Thanks for voting for {userVoted}! ðŸŽ‰
          </p>
        </div>
      )}
    </div>
  );
};

export const SimpleChart = () => {
